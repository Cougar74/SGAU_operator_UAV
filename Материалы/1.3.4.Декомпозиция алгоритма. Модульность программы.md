    def is_safe(board, row, col):
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –≤–µ—Ä—Ç–∏–∫–∞–ª—å –∏ –¥–∏–∞–≥–æ–Ω–∞–ª–∏

        return True

    def backtrack(col, current_board, solutions):
        # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: –≤—Å–µ —Ñ–µ—Ä–∑–∏ —Ä–∞—Å—Å—Ç–∞–≤–ª–µ–Ω—ã

        # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ —Å—Ç—Ä–æ–∫–∏ –¥–ª—è —Ç–µ–∫—É—â–µ–≥–æ —Å—Ç–æ–ª–±—Ü–∞

                # –î–æ–±–∞–≤–ª—è–µ–º –∫–∞–Ω–¥–∏–¥–∞—Ç–∞

                # –∞ current_board[col] –ø–µ—Ä–µ–∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è)

    solutions = []
    # board[i] = j –æ–∑–Ω–∞—á–∞–µ—Ç, —á—Ç–æ –≤ —Å—Ç–æ–ª–±—Ü–µ i —Ñ–µ—Ä–∑—å —Å—Ç–æ–∏—Ç –≤ —Å—Ç—Ä–æ–∫–µ j
    board = [-1] * n

    return solutions

# –ù–∞–π–¥–µ–º –≤—Å–µ —Ä–µ—à–µ–Ω–∏—è –¥–ª—è –¥–æ—Å–∫–∏ 4x4

import sys

def tsp_backtracking(distances):
    n = len(distances)
    best_path = None
    min_cost = sys.maxsize
    
    def backtrack(current_path, current_cost, visited):
        nonlocal best_path, min_cost
        
        # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: –≤—Å–µ –≥–æ—Ä–æ–¥–∞ –ø–æ—Å–µ—â–µ–Ω—ã

            # –í–æ–∑–≤—Ä–∞—â–∞–µ–º—Å—è –≤ –Ω–∞—á–∞–ª—å–Ω—ã–π –≥–æ—Ä–æ–¥

            total_cost = current_cost + return_cost
            
            if total_cost < min_cost:
                min_cost = total_cost
                best_path = current_path[:] + [current_path[0]]  # –î–æ–±–∞–≤–ª—è–µ–º –≤–æ–∑–≤—Ä–∞—Ç –≤ –Ω–∞—á–∞–ª–æ

        # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ –Ω–µ–ø–æ—Å–µ—â–µ–Ω–Ω—ã–µ –≥–æ—Ä–æ–¥–∞

            if not visited[next_city]:
                # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º —Å—Ç–æ–∏–º–æ—Å—Ç—å –ø–µ—Ä–µ—Ö–æ–¥–∞ –∫ —Å–ª–µ–¥—É—é—â–µ–º—É –≥–æ—Ä–æ–¥—É

                else:
                    new_cost = 0  # –ù–∞—á–∏–Ω–∞–µ–º –ø—É—Ç—å
                
                # –û–¢–°–ï–ß–ï–ù–ò–ï: –µ—Å–ª–∏ —Ç–µ–∫—É—â–∞—è —Å—Ç–æ–∏–º–æ—Å—Ç—å —É–∂–µ –ø—Ä–µ–≤—ã—à–∞–µ—Ç –ª—É—á—à—É—é –Ω–∞–π–¥–µ–Ω–Ω—É—é

                current_path.append(next_city)

                backtrack(current_path, new_cost, visited)

                current_path.pop()
                visited[next_city] = False
    
    # –ù–∞—á–∏–Ω–∞–µ–º –ø–æ–∏—Å–∫ —Å –ø—É—Å—Ç–æ–≥–æ –º–∞—Ä—à—Ä—É—Ç–∞

    backtrack([start_city], 0, visited)
    
    return best_path, min_cost
